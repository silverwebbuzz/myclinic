Key rule: The clinic owner/admin (the person who signed up) is always free and never counts as a seat. Every additional user — doctor, nurse, receptionist, lab tech — counts as 1 seat. Patient portal users never count as seats.

Custom permission overrides (admin can fine-tune)
Beyond the role defaults, admin can toggle specific permissions per staff member — stored in users.custom_permissions JSON. Example: give the receptionist access to view (but not edit) clinical notes. Or restrict a doctor to only their own patients (useful for multi-doctor clinics where doctors should not see each other's patients). The permission check: RbacService::can($user, 'patients.view_others') → checks role default → then checks custom_permissions override.

CREATE TABLE staff_invitations (
  id              BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  clinic_id       BIGINT UNSIGNED NOT NULL,
  invited_by      BIGINT UNSIGNED NOT NULL,
  name            VARCHAR(100) NOT NULL,
  email           VARCHAR(150) NOT NULL,
  role            ENUM('doctor','nurse','receptionist','labtech') NOT NULL,
  token_hash      VARCHAR(64) NOT NULL,  -- hash of the token in the email link
  expires_at      TIMESTAMP NOT NULL,    -- 72 hours from created_at
  accepted_at     TIMESTAMP NULL,
  created_user_id BIGINT UNSIGNED NULL,  -- set after acceptance
  status          ENUM('pending','accepted','expired','revoked') DEFAULT 'pending',
  created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  KEY idx_clinic_invites (clinic_id, status),
  KEY idx_email (email, status),
  FOREIGN KEY (clinic_id)       REFERENCES tenants(id) ON DELETE CASCADE,
  FOREIGN KEY (invited_by)      REFERENCES users(id),
  FOREIGN KEY (created_user_id) REFERENCES users(id) ON DELETE SET NULL
) ENGINE=InnoDB;
-- SeatService PHP logic (add to /app/Services/SeatService.php)
// SeatService.php
class SeatService {

  public function canAddStaff(int $clinicId): bool {
    $limit   = $this->getEffectiveLimit($clinicId);
    $used    = $this->getSeatUsage($clinicId);
    return $used < $limit;
  }

  public function getEffectiveLimit(int $clinicId): int {
    $tenant = QB::table('tenants')->find($clinicId);
    return $tenant->seat_limit + $tenant->extra_seats_purchased;
  }

  public function getSeatUsage(int $clinicId): int {
    // Active non-owner staff + pending invitations (seat reserved)
    $activeStaff = QB::table('users')
      ->forClinic($clinicId)
      ->where('is_owner', 0)
      ->where('is_active', 1)
      ->count();

    $pendingInvites = QB::table('staff_invitations')
      ->forClinic($clinicId)
      ->where('status', 'pending')
      ->where('expires_at', '>', now())
      ->count();

    return $activeStaff + $pendingInvites;
  }

  public function sendInvite(int $clinicId, int $invitedBy,
                              string $name, string $email,
                              string $role): array {
    if (!$this->canAddStaff($clinicId)) {
      throw new SeatLimitException('Seat limit reached. Upgrade plan or purchase extra seats.');
    }

    $token     = bin2hex(random_bytes(32));  // 64-char token
    $tokenHash = hash('sha256', $token);

    QB::table('staff_invitations')->insert([
      'clinic_id'  => $clinicId,
      'invited_by' => $invitedBy,
      'name'       => $name,
      'email'      => $email,
      'role'       => $role,
      'token_hash' => $tokenHash,
      'expires_at' => date('Y-m-d H:i:s', time() + 72 * 3600),
      'status'     => 'pending',
    ]);

    // Send email via Mailgun
    MailService::send($email, 'staff_invite', [
      'name'       => $name,
      'role'       => ucfirst($role),
      'clinic'     => $clinicId,
      'invite_url' => "https://app.manageclinic.com/accept-invite/{$token}",
    ]);

    return ['success' => true, 'seats_remaining' => $this->getEffectiveLimit($clinicId) - $this->getSeatUsage($clinicId) - 1];
  }

  public function acceptInvite(string $token, string $password): User {
    $tokenHash = hash('sha256', $token);
    $invite    = QB::table('staff_invitations')
      ->where('token_hash', $tokenHash)
      ->where('status', 'pending')
      ->where('expires_at', '>', now())
      ->first();

    if (!$invite) throw new InvalidTokenException('Invalid or expired invite link.');

    // Create the user account
    $userId = QB::table('users')->insert([
      'clinic_id'     => $invite->clinic_id,
      'name'          => $invite->name,
      'email'         => $invite->email,
      'role'          => $invite->role,
      'is_owner'      => 0,
      'password_hash' => password_hash($password, PASSWORD_BCRYPT),
      'is_active'     => 1,
    ]);

    // Update invitation record
    QB::table('staff_invitations')->where('id', $invite->id)->update([
      'status'          => 'accepted',
      'accepted_at'     => now(),
      'created_user_id' => $userId,
    ]);

    return QB::table('users')->find($userId);
  }
}
